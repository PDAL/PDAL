
add_subdirectory(libE57Format)

PDAL_ADD_PLUGIN(e57plugin_reader reader e57
    FILES
        io/E57Reader.cpp
        io/Scan.cpp
        io/Utils.cpp
    INCLUDES
        ${NLOHMANN_INCLUDE_DIR}
        ${PDAL_VENDOR_DIR}
    LINK_WITH
        E57Format
    INCLUDES
        libE57Format/include
)
target_compile_definitions(${e57plugin_reader}	
    PRIVATE
        ARBITER_ZLIB
        ARBITER_DLL_IMPORT
)

    FILES
        io/E57Writer.cpp
        io/Utils.cpp
        io/Uuid.cpp
    LINK_WITH
        E57Format
    INCLUDES
        libE57Format/include
)
